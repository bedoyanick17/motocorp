Código Java

Ejemplo del código desarrollado en Java para MotoCorp.


import java.util.Scanner;

public class CiclosWhile {

    public static void main(String[] args) {

        Scanner nk = new Scanner(System.in);

        double[] temperaturas = new double[7];
        int contador = 0;
        String respuesta;


        while (contador < temperaturas.length) {
            System.out.print("Ingresa la temperatura del día " + (contador + 1) + ": ");
            temperaturas[contador] = nk.nextDouble();
            contador++;


            if (contador == temperaturas.length) {
                System.out.println("Se han registrado todas las temperaturas de la semana.");
                break;
            }


            System.out.print("¿Deseas ingresar otra temperatura? (s/n): ");
            respuesta = nk.next().toLowerCase();

            if (respuesta.equals("n")) {
                break;
            }
        }


        System.out.println("\nTemperaturas registradas:");
        for (int i = 0; i < contador; i++) {
            System.out.println("Día " + (i + 1) + ": " + temperaturas[i] + "°C");
        }


        if (contador > 0) {
            double max = temperaturas[0];
            double min = temperaturas[0];

            for (int i = 1; i < contador; i++) {
                if (temperaturas[i] > max) {
                    max = temperaturas[i];
                }
                if (temperaturas[i] < min) {
                    min = temperaturas[i];
                }
            }

            System.out.println("\nTemperatura más alta: " + max + "°C");
            System.out.println("Temperatura más baja: " + min + "°C");
        } else {
            System.out.println("No se registraron temperaturas.");
        }

    }
}
